00162 void DmaIrqHandler(uint8_t u8DmaChannel)
00163 {
00164   uint32_t u32DmaRegisterBaseAddress;
00165   uint32_t u32RegisterReadOut;
00166   uint8_t  u8StopStatus;
00167   uint8_t u8InstanceIdx;
00168   
00169   u32DmaRegisterBaseAddress = GET_DMA_ADDRESS(u8DmaChannel);
00170   
00171   // Readout DMACB and update
00172   u32RegisterReadOut = *(uint32_t*)((uint32_t)(u32DmaRegisterBaseAddress + DMA_DMACB_OFFSET));
00173   
00174   u8StopStatus = (uint8_t)((u32RegisterReadOut & 0x00070000ul)>> 16u);
00175 
00176   u8InstanceIdx = DmaGetInternDataIdx(u8DmaChannel); // Get DMA instance index
00177   if(255u == u8InstanceIdx) // Instance not found.
00178   {
00179     return;
00180   }
00181   
00182   // Clear interrupt cause
00183   u32RegisterReadOut &= 0xFFF8FFFFu;     
00184   *(uint32_t*)(uint32_t)((u32DmaRegisterBaseAddress + DMA_DMACB_OFFSET)) 
00185     = u32RegisterReadOut;  
00186   
00187   if (0x5u == u8StopStatus)  // Successful DMA transfer
00188   {
00189     stcDmaInternData.apfnDmaIrqCompletionCb[u8InstanceIdx](); 
00190   }
00191   else                      // Errornous DMA transfer
00192   {
00193     stcDmaInternData.apfnDmaErrorIrqCb[u8InstanceIdx](u8StopStatus);
00194   }
