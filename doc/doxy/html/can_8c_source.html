00314                 unCanIf2Cmsk.MASK     = 1u; // Transfer Identifier Mask + MDir + MXtd to Message Object
00315                 unCanIf2Cmsk.ARB      = 1u; // Transfer Identifier + Dir + Xtd + MsgVal to Message Object

00320                 unCanIf2Cmsk.DATAB    = 1u; // Transfer Data Bytes 4-7 to Mesage Object
00321                 pstcCan->IF2CMSK_f = unCanIf2Cmsk;
00322 
00323                 // Write message buffer number to request reading
00324                 // the message buffer into the interface registers
00325                 // We can write full 16bit here because BUSY bit is not used in this mode
00326                 pstcCan->IF2CREQ = u16MessageBufferNumber ;
00327 
00328                 // Check if message object is configured for reception or transmission
00329                 if (CAN_MSG_DIR_RX == pstcCan->IF2ARB_f.DIR)
00330                 {
00331                     // CAN message received, check if there is a callback function
00332                     // for this message buffer registered
00333                     if (NULL != pstcCanInternData->stcCanInterruptHandling[u16MessageBufferNumber - 1].pstcMsgBuffer)
00334                     {
00335                         stc_can_msg_t* pstcMsgBuffer = pstcCanInternData->stcCanInterruptHandling[u16MessageBufferNumber - 1].pstcMsgBuffer;
00336                         uint16_t* pu16Data = (uint16_t*)&pstcMsgBuffer->stcData.au8Data[0] ;
00337 
00338                         // Check if 11- or 29-bit identifier was used for message.
00339                         if ( CAN_MSG_ID_EXTENDED != pstcCan->IF2ARB2_f.XTD )
00340                         {
00341                             // 11-bit identifier was used.
00342                             pstcMsgBuffer->stcIdentifier.bExtended = FALSE;
00343                             pstcMsgBuffer->stcIdentifier.u32Identifier = (pstcCan->IF2ARB_f.ID >> 18);
00344                         }
00345                         else
00346                         {
00347                             // 29-bit identifier was used.
00348                             pstcMsgBuffer->stcIdentifier.bExtended = TRUE;
00349                             pstcMsgBuffer->stcIdentifier.u32Identifier = pstcCan->IF2ARB_f.ID;
00350                         }
00351

00353                         pu16Data[0] = pstcCan->IF2DTA1_L; 
00354                         pu16Data[1] = pstcCan->IF2DTA2_L;
00355                         pu16Data[2] = pstcCan->IF2DTB1_L; 
00356                         pu16Data[3] = pstcCan->IF2DTB2_L; 
00357 
00358                         // Set data length of received data bytes.
00359                         pstcMsgBuffer->stcData.u8DataLengthCode = pstcCan->IF2MCTR_f.DLC;
00360 
00361                         // Check if message lost flag is set in receive message buffer.
00362                         if ( 1u == pstcCan->IF2MCTR_f.MSGLST )
00363                         {
00364                             pstcMsgBuffer->bOverflow = TRUE ;
00365 
00366                             // Reset message lost flag.
00367                             pstcCan->IF2MCTR_f.MSGLST = 0 ;
00368 
00369                             // Write command mask register to perform a write to CAN control register.
00370                             // Pls. note that INTPND and NEWDAT are left unchanged if written with '0'
00371                             unCanIf2Cmsk = pstcCan->IF2CMSK_f;
00372                             unCanIf2Cmsk.WRRD     = CAN_MSG_DIR_RX; // Set direction to tx
00373                             unCanIf2Cmsk.CONTROL  = 1;              // Transfer control bits
00374                             pstcCan->IF2CMSK_f = unCanIf2Cmsk;
00375 
00376                             // Set message buffer number to update message buffer.
00377                             // We can write full 16bit here because BUSY bit is not used in this mode
00378                             pstcCan->IF2CREQ = u16MessageBufferNumber;
00379                         }
00380 
00381                         // Check if data new flag is already set.
00382                         if ( TRUE == pstcMsgBuffer->bNew )
