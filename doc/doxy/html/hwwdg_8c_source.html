00178 en_result_t Hwwdg_DeInit(uint32_t u32MagicWord1,
00179                          uint32_t u32MagicWord2
00180                         )
00181 {
00182     en_result_t enResult;
00183 
00184     enResult = ErrorInvalidParameter;
00185 
00186     /* Inverted magic word check is done to avoid "plain text magic word" in ROM. */
00187     if ((HWWDG_MAGIC_WORD_CHK_RESULT == (u32MagicWord1 ^ HWWDG_MAGIC_WORD_CHK1)) &&
00188         (HWWDG_MAGIC_WORD_CHK_RESULT == (u32MagicWord2 ^ HWWDG_MAGIC_WORD_CHK2))
00189        )
00190     {
00191         /* HW Watchdog Control Register unlock sequence */
00192         FM_HWWDT->WDG_LCK = HWWDG_REG_UNLOCK_1;
00193         FM_HWWDT->WDG_LCK = HWWDG_REG_UNLOCK_2;
00194 
00195         /* HW Watchdog stop */
00196         FM_HWWDT->WDG_CTL = 0;
00197 
00198         /* Initialize callback to NULL */
00199         pfnHwwdgCallback = NULL;
00200 
00201         enResult = Ok;
00202     }
