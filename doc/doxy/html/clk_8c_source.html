00480 en_result_t Clk_SetSource(en_clk_source_t enSource)
00481 {  
00482   uint8_t u8Rcs, u8Rcm;
00483   switch(enSource)
00484   {
00485     case ClkMain:
00486       if ((TRUE != FM_CRG->SCM_CTL_f.MOSCE) ||  // Main Oscillator ready?
00487           (TRUE != FM_CRG->SCM_STR_f.MORDY))
00488       {
00489         return ErrorInvalidMode ;
00490       }
00491       FM_CRG->SCM_CTL_f.RCS = 0x1u;
00492       break;
00493     case ClkSub:
00494       if ((TRUE != FM_CRG->SCM_CTL_f.SOSCE) ||  // Sub Oscillator ready?
00495           (TRUE != FM_CRG->SCM_STR_f.SORDY))
00496       {
00497         return ErrorInvalidMode ;
00498       }
00499       FM_CRG->SCM_CTL_f.RCS = 0x5u;
00500       break;
00501     case ClkHsCr:                                // Always possible
00502       FM_CRG->SCM_CTL_f.RCS = 0x0u;
00503       break;     
00504     case ClkLsCr:                                // Always possible
00505       FM_CRG->SCM_CTL_f.RCS = 0x4u;
00506       break;
00507     case ClkHsCrPll:  
00508       FM_CRG->PSW_TMR_f.PINC = 1u;
00509       FM_CRG->SCM_CTL_f.RCS = 0x2u;
00510       break;
00511     case ClkPll:                           
00512       if ((TRUE != FM_CRG->SCM_STR_f.MORDY) ||  // PLL ready?
00513           (TRUE != FM_CRG->SCM_STR_f.PLRDY))
00514       {
00515         return ErrorInvalidMode ;
00516       }
00517       
00518       FM_CRG->PSW_TMR_f.PINC = 0u;
00519       FM_CRG->SCM_CTL_f.RCS = 0x2u;
00520       break;
00521     default:
00522       return ErrorInvalidParameter ;
00523   }
00524 
00525   /* Wait until switch stable */
00526   while(1)
00527   {
00528     u8Rcs = FM_CRG->SCM_CTL_f.RCS;
00529     u8Rcm = FM_CRG->SCM_STR_f.RCM;
00530     if(u8Rcs == u8Rcm)
00531     {
00532         break;
00533     }
00534   }
00535 
00536   return Ok;
00537 } // Clk_SetSource
