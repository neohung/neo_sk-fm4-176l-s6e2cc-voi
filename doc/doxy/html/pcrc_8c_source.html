00170 en_result_t Pcrc_Init(stc_pcrc_config_t* pstcConfig)
00171 {  
00172     en_result_t enResult;
00173     stc_pcrc_prgcrc_cfg_field_t PRGCRC_CFG_f;
00174 
00175     PDL_ZERO_STRUCT(PRGCRC_CFG_f);
00176 
00177     /* Check for valid pointers and configrue parameters */
00178     if ((NULL == pstcConfig) ||
00179         (pstcConfig->enInputFormat > LsbFirstLittleEndian) ||
00180         (pstcConfig->enOutputFormat > LsbFirstLittleEndian) ||
00181         (pstcConfig->enInputDataSize > InputData32Bit))
00182     {
00183         enResult = ErrorInvalidParameter;
00184     }
00185     else
00186     {
00187         PRGCRC_CFG_f.FI = pstcConfig->enInputFormat;
00188         PRGCRC_CFG_f.FO = pstcConfig->enOutputFormat;
00189         PRGCRC_CFG_f.SZ = pstcConfig->enInputDataSize;
00190         PRGCRC_CFG_f.TEST = 0x20u;              //Always writes "100000" to perform write access to this register.
00191 
00192         /* Update hardware */
00193         FM_PCRC->PRGCRC_POLY = pstcConfig->u32GeneratorPolynomial;
00194         FM_PCRC->PRGCRC_SEED = pstcConfig->u32CrcInitValue;
00195         FM_PCRC->PRGCRC_FXOR = pstcConfig->u32FinalXorValue;
00196         FM_PCRC->PRGCRC_CFG_f = PRGCRC_CFG_f;
00197         FM_PCRC->PRGCRC_CFG_f.CDEN = ((pstcConfig->bUseDstc == TRUE) ? 1u : 0u);
00198 
00199 #if (PDL_INTERRUPT_ENABLE_PCRC== PDL_ON)
00200         m_pfnPcrcIrqCb = pstcConfig->pfnIrqCb;
00201         FM_PCRC->PRGCRC_CFG_f.CIRQCLR = 1u;
00202         FM_PCRC->PRGCRC_CFG_f.CIEN = ((pstcConfig->bIrqEn == TRUE) ? 1u : 0u);
00203          // Initialize NVIC
00204         if(TRUE == pstcConfig->bTouchNvic)
00205         {
00206             PcrcInitNvic();
00207         }
00208 #endif        
00209         enResult = Ok;
00210     }
00211 
00212     return (enResult);
